\newcommand{\psalterday}[2]{
    \section{Day #1. #2 Prayer}
    \directlua{
        if "#2" == "Morning" then
            tex.print([[\noexpand\renewcommand{\noexpand\daynumber}{Day #1: M.}]])
        else 
        tex.print([[\noexpand\renewcommand{\noexpand\daynumber}{Day #1: E.}]])
        end
    }
}
